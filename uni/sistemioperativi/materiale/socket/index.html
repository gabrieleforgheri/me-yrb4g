1. FUNZIONI DI I/O SICURE (Full Read & Full Write)

I file descriptor delle socket possono leggere o scrivere meno byte di quelli richiesti (partial read/write) oppure essere interrotti da segnali (EINTR). All'esame devi usare dei loop protetti come questi:

Scrittura Sicura (write_full)

static ssize_t write_full(int fd, const void *buf, size_t count) {
    size_t done = 0;
    const char *p = buf; // Puntatore per l'aritmetica dei byte

    while (done < count) {
        ssize_t n = write(fd, p + done, count - done); // Scrive i byte rimanenti
        if (n < 0) {
            if (errno == EINTR) continue; // Interrotto da segnale, riprova
            return -1; // Vero errore
        }
        done += (size_t)n; // Aggiorna il contatore
    }
    return (ssize_t)done;
}

Lettura Sicura (read_full)

static ssize_t read_full(int fd, void *buf, size_t count) {
    size_t done = 0;
    char *p = buf; // Puntatore per l'aritmetica dei byte

    while (done < count) {
        ssize_t n = read(fd, p + done, count - done); // Legge i byte mancanti
        if (n == 0) return 0;  // Il peer ha chiuso la connessione (EOF)
        if (n < 0) {
            if (errno == EINTR) continue; // Interrotto da segnale, riprova
            return -1; // Vero errore
        }
        done += (size_t)n; // Aggiorna il contatore
    }
    return (ssize_t)done;
}

2. LATO SERVER (Il processo in ascolto)

Il Server crea la socket, la lega a un percorso sul file system, si mette in ascolto e genera un nuovo processo figlio per ogni client connesso (Server Concorrente).

#define SOCKET_PATH "/tmp/concurrent.sock"

// 1. Creazione della socket
int srv_fd = socket(AF_UNIX, SOCK_STREAM, 0);
if (srv_fd == -1) { perror("socket"); exit(1); }

// 2. Rimozione di eventuali socket file residui 
unlink(SOCKET_PATH);

// 3. Preparazione dell'indirizzo
struct sockaddr_un srv_addr;
memset(&srv_addr, 0, sizeof(srv_addr));
srv_addr.sun_family = AF_UNIX;
strncpy(srv_addr.sun_path, SOCKET_PATH, sizeof(srv_addr.sun_path) - 1);

// 4. Bind (Collegamento file-socket)
if (bind(srv_fd, (struct sockaddr *)&srv_addr, sizeof(srv_addr)) == -1) {
    perror("bind"); exit(1);
}

// 5. Listen (Mette la socket in modalitÃ  passiva/ascolto, backlog 10)
if (listen(srv_fd, 10) == -1) {
    perror("listen"); exit(1);
}

// 6. Ciclo di accettazione e Server Concorrente
while (1) {
    struct sockaddr_un cli_addr;
    int addr_size = sizeof(cli_addr);
    
    // Accept crea un NUOVO file descriptor (conn_fd) dedicato a QUESTO client
    int conn_fd = accept(srv_fd, (struct sockaddr *)&cli_addr, &addr_size);
    if (conn_fd == -1) {
        if (errno == EINTR) continue; // Segnale durante l'attesa
        break; // Errore critico
    }

    pid_t child_pid = fork();
    if (child_pid == 0) { // FIGLIO (Helper)
        close(srv_fd); // Il figlio non deve accettare nuove connessioni!
        
        // Esegue il lavoro per il client leggendo/scrivendo su conn_fd
        // Esempio: read_full(conn_fd, &request, sizeof(request));
        
        close(conn_fd); // Chiude la sua connessione
        _exit(0); // Termina il processo helper
    }
    
    // PADRE
    close(conn_fd); // Il padre chiude il FD del client per non esaurire le risorse
}

// Pulizia Server
close(srv_fd);
unlink(SOCKET_PATH);

3. LATO CLIENT (Il processo che si connette)

Il Client si connette semplicemente al SOCKET_PATH dove il server Ã¨ in ascolto.

// 1. Creazione della socket

if (sock_fd == -1) { perror("socket"); exit(1); }

// 2. Preparazione dell'indirizzo del Server

memset(&srv_addr, 0, sizeof(srv_addr));
srv_addr.sun_family = AF_UNIX;
strncpy(srv_addr.sun_path, SOCKET_PATH, sizeof(srv_addr.sun_path) - 1);

// 3. Connessione
if (connect(sock_fd, (struct sockaddr *)&srv_addr, sizeof(srv_addr)) == -1) {
    perror("connect");
    close(sock_fd);
    exit(1);
}

Note Fondamentali per l'esame:

  • Differenza tra Descrittori: Sul server ricorda sempre che srv_fd serve esclusivamente per rimanere in ascolto (tramite accept), mentre la comunicazione reale coi client avviene su conn_fd (che Ã¨ generato proprio dall'accept).
  • Unlink Preventivo: Non dimenticare l'unlink(SOCKET_PATH) prima di fare la bind sul server. Se il server precedente si era chiuso male, il file sul file system bloccherÃ  le nuove aperture.
  • Tipi di Dati: Sulle socket si inviano spesso strutture struct complete (usando i puntatori e sizeof), proprio come si faceva con le pipe.
